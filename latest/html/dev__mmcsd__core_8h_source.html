186rt_inline rt_uint32_t __rt_fls(rt_uint32_t val)
187{
188 rt_uint32_t bit = 32;
189
190 if (!val)
191 return 0;
192 if (!(val & 0xffff0000u))
193 {
194 val <<= 16;
195 bit -= 16;
196 }
197 if (!(val & 0xff000000u))
198 {
199 val <<= 8;
200 bit -= 8;
201 }
202 if (!(val & 0xf0000000u))
203 {
204 val <<= 4;
205 bit -= 4;
206 }
207 if (!(val & 0xc0000000u))
208 {
209 val <<= 2;
210 bit -= 2;
211 }
212 if (!(val & 0x80000000u))
213 {
214 bit -= 1;
215 }
216
217 return bit;
218}
